rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Fallback: deny all by default
    match /{document=**} {
      allow read, write: if false;
    }

    // Owner-based access to users collection and all of its descendant subcollections (notes, reminders, etc.)
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
      
      match /{allSubcollections=**} {
        allow read, write: if request.auth != null && request.auth.uid == userId;
      }
    }

    // Global public creator orders readable by everyone, but only writable by Carlos Delgado admin email
    match /creator_orders/{orderId} {
      allow read: if true;
      allow write: if request.auth != null && (
        request.auth.token.email == "carlosdelgado.neska@gmail.com" ||
        request.auth.token.email == "carlosdelgadoneska@gmail.com"
      );
    }
  }
}
